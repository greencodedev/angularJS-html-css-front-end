create table Accounts (
AccountId int AUTO_INCREMENT PRIMARY KEY,
CompanyName varchar(255),
Email varchar(255),
FirstName varchar(255),
LastName varchar(255),
Phone varchar(255),
Address varchar(255),
Status varchar(45)
);
